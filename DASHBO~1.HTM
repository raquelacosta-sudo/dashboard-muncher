📋 Query de referencia — Operativa
SELECT A.BRAND_ID, COUNT(DISTINCT A.ORDER_ID) AS ORDERS, SUM(CASE WHEN A.IS_DEFECT=TRUE THEN 1 ELSE 0 END) AS DEFECT_ORDERS, ROUND(SUM(CASE WHEN A.IS_DEFECT=TRUE THEN 1 ELSE 0 END)*100.0/COUNT(DISTINCT A.ORDER_ID),2) AS DEFECT_RATE, ROUND(AVG(A.PREPARATION_TIME)/60.0,1) AS AVG_PREP_MIN, ROUND(AVG(A.TO_USER_TIME)/60.0,1) AS AVG_DELIVERY_MIN, ROUND(AVG(A.TOTAL_ORDER_TIME)/60.0,1) AS AVG_TOTAL_MIN FROM RP_GOLD_DB_PROD.DES_PROD.ORDER_DIMENSIONS A JOIN RP_GOLD_DB_PROD.RESTAURANTES_GLOBAL_MDA.A_SCORECARD_REST4 B ON A.STORE_ID=B.STORE_ID WHERE A.COUNTRY='MX' AND A.COUNT_TO_GMV=TRUE AND A.CREATED_AT >= '2025-12-10' AND A.CREATED_AT < '2026-04-23' AND B.BRAND_ID IN (361448,365632,484188,504732,575113,542049,543391,542042,575458,575455) GROUP BY 1 ORDER BY 1

📋 Query: Retención y Recompra 30/60/90 días
WITH first_order AS ( SELECT A.APPLICATION_USER_ID, A.BRAND_ID, B.BRAND_NAME, MIN(A.CREATED_AT) AS FIRST_ORDER_DT FROM RP_GOLD_DB_PROD.DES_PROD.ORDER_DIMENSIONS A JOIN RP_GOLD_DB_PROD.RESTAURANTES_GLOBAL_MDA.A_SCORECARD_REST4 B ON A.STORE_ID = B.STORE_ID WHERE A.COUNTRY = 'MX' AND A.COUNT_TO_GMV = TRUE AND A.BRAND_ID IN (361448,365632,575113,504732, 484188,542049,543391,542042,575458,575455,612185) AND A.CREATED_AT >= '2025-12-10' GROUP BY 1,2,3 ), recompra AS ( SELECT F.APPLICATION_USER_ID, F.BRAND_ID, F.BRAND_NAME, MAX(CASE WHEN O.CREATED_AT > F.FIRST_ORDER_DT AND O.CREATED_AT <= DATEADD('day',30,F.FIRST_ORDER_DT) THEN 1 ELSE 0 END) AS BOUGHT_30D, MAX(CASE WHEN O.CREATED_AT > F.FIRST_ORDER_DT AND O.CREATED_AT <= DATEADD('day',60,F.FIRST_ORDER_DT) THEN 1 ELSE 0 END) AS BOUGHT_60D, MAX(CASE WHEN O.CREATED_AT > F.FIRST_ORDER_DT AND O.CREATED_AT <= DATEADD('day',90,F.FIRST_ORDER_DT) THEN 1 ELSE 0 END) AS BOUGHT_90D, COUNT(DISTINCT O.ORDER_ID) AS TOTAL_ORDERS FROM first_order F JOIN RP_GOLD_DB_PROD.DES_PROD.ORDER_DIMENSIONS O ON F.APPLICATION_USER_ID = O.APPLICATION_USER_ID AND F.BRAND_ID = O.BRAND_ID AND O.COUNTRY = 'MX' AND O.COUNT_TO_GMV = TRUE GROUP BY 1,2,3 ) SELECT BRAND_ID, BRAND_NAME, COUNT(DISTINCT APPLICATION_USER_ID) AS UNIQUE_USERS, SUM(BOUGHT_30D) AS RECOMPRA_30D, ROUND(SUM(BOUGHT_30D)*100.0/COUNT(*),1) AS PCT_30D, SUM(BOUGHT_60D) AS RECOMPRA_60D, ROUND(SUM(BOUGHT_60D)*100.0/COUNT(*),1) AS PCT_60D, SUM(BOUGHT_90D) AS RECOMPRA_90D, ROUND(SUM(BOUGHT_90D)*100.0/COUNT(*),1) AS PCT_90D, ROUND(SUM(TOTAL_ORDERS)*1.0/COUNT(*),2) AS AVG_ORDERS_PER_USER FROM recompra GROUP BY 1,2 ORDER BY 1
